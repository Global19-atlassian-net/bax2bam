include_directories(
    .
    ${BLASR_INCLUDE_DIRS}
    ${Boost_INCLUDE_DIRS}
    ${HDF5_INCLUDE_DIRS}
    ${PacBioBAM_INCLUDE_DIRS}
    ${PBDATA_INCLUDE_DIRS}
    ${PBDATA_ROOT_DIR}
    ${PBIHDF_INCLUDE_DIRS}
    ${ZLIB_INCLUDE_DIRS}
)

    main.cpp
    Bax2Bam.h
    Bax2Bam.cpp
    CcsConverter.h
    CcsConverter.cpp
    ConverterBase.h
    HqRegionConverter.h
    HqRegionConverter.cpp
    IConverter.h
    IConverter.cpp
    OptionParser.h
    OptionParser.cpp
    PolymeraseReadConverter.h
    PolymeraseReadConverter.cpp
    Settings.h
    Settings.cpp
    SubreadConverter.h
    SubreadConverter.cpp

    ${PBBAM_LINK_FLAG}

    ${PBIHDF_LIBRARIES}
    ${PBDATA_LIBRARIES}
    ${HDF5_CPP_LIBRARIES}
    ${HDF5_LIBRARIES}
    ${PacBioBAM_LIBRARIES}
    ${HTSLIB_LDFLAGS}
    ${ZLIB_LIBRARIES}
    ${CMAKE_THREAD_LIBS_INIT}
    ${MY_LIBRT}
    "-ldl"
